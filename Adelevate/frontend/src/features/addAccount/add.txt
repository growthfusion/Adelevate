import React, { useState, useRef, useEffect } from "react";
import nb from "@/assets/images/automation_img/NewsBreak.svg";
import fb from "@/assets/images/automation_img/Facebook.svg";
import snapchatIcon from "@/assets/images/automation_img/snapchat.svg";
import tiktokIcon from "@/assets/images/automation_img/tiktok.svg";
import googleIcon from "@/assets/images/automation_img/google.svg";
import { addAccount, accountExists } from "@/services/accountsConfig";

function Add_Acounts() {
  const [selectedPlatform, setSelectedPlatform] = useState("");
  const [isOpen, setIsOpen] = useState(false);
  const [bmName, setBmName] = useState("");
  const [accessToken, setAccessToken] = useState("");
  const [accountId, setAccountId] = useState("");
  const [accountLabel, setAccountLabel] = useState("");
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [error, setError] = useState("");
  const [success, setSuccess] = useState("");
  const dropdownRef = useRef(null);

  const platforms = [
    { name: "Meta", icon: fb, value: "facebook" },
    { name: "NewsBreak", icon: nb, value: "newsbreak" },
    { name: "Snapchat", icon: snapchatIcon, value: "snapchat" },
    { name: "TikTok", icon: tiktokIcon, value: "tiktok" },
    { name: "Google", icon: googleIcon, value: "google" },
  ];

  const handleSubmit = async (e) => {
    e.preventDefault();
    setError("");
    setSuccess("");
    setIsSubmitting(true);

    try {
      const accountData = {
        platform: selectedPlatform,
        ...(selectedPlatform === "facebook" && { bmName }),
        accessToken,
        accountId,
        accountLabel,
      };

      console.log("ðŸ“¤ Sending to Firestore:", accountData);

      // Check if account already exists
      const exists = await accountExists(selectedPlatform, accountId);
      if (exists) {
        setError("An account with this ID already exists for this platform.");
        setIsSubmitting(false);
        return;
      }

      // Add account to Firestore
      const id = await addAccount(accountData);
      console.log("âœ… Account added successfully with ID:", id);

      setSuccess("Account added successfully!");

      // Reset form after 2 seconds
      setTimeout(() => {
        setSelectedPlatform("");
        setBmName("");
        setAccessToken("");
        setAccountId("");
        setAccountLabel("");
        setSuccess("");
      }, 2000);
    } catch (err) {
      console.error("âŒ Error adding account:", err);
      setError(err.message || "Failed to add account. Please try again.");
    } finally {
      setIsSubmitting(false);
    }
  };

  const getSelectedPlatform = () => {
    return platforms.find((p) => p.value === selectedPlatform);
  };

  const handleSelectPlatform = (value) => {
    setSelectedPlatform(value);
    setIsOpen(false);
    setError("");
  };

  // Close dropdown when clicking outside
  useEffect(() => {
    const handleClickOutside = (event) => {
      if (dropdownRef.current && !dropdownRef.current.contains(event.target)) {
        setIsOpen(false);
      }
    };

    document.addEventListener("mousedown", handleClickOutside);
    return () => document.removeEventListener("mousedown", handleClickOutside);
  }, []);

  return (
    <div className="min-h-screen bg-gray-50 flex items-start justify-center max-sm:pt-[20%] pt-0">
      <div className="w-full p-3 sm:p-4 md:p-6 lg:p-8">
        <div className="w-full max-w-md sm:max-w-lg md:max-w-xl lg:max-w-2xl xl:max-w-3xl 2xl:max-w-4xl mx-auto">
          <h1 className="text-lg sm:text-xl md:text-2xl lg:text-2xl xl:text-3xl font-bold mb-3 sm:mb-4 md:mb-5 lg:mb-6 text-gray-800 text-center sm:text-left">
            Integration
          </h1>

          {/* Success Message */}
          {success && (
            <div className="mb-4 p-3 sm:p-4 bg-green-50 border border-green-200 text-green-700 rounded-lg flex items-center gap-2 animate-fadeIn">
              <svg
                className="w-5 h-5 flex-shrink-0"
                fill="currentColor"
                viewBox="0 0 20 20"
              >
                <path
                  fillRule="evenodd"
                  d="M10 18a8 8 0 100-16 8 8 0 000 16zm3.707-9.293a1 1 0 00-1.414-1.414L9 10.586 7.707 9.293a1 1 0 00-1.414 1.414l2 2a1 1 0 001.414 0l4-4z"
                  clipRule="evenodd"
                />
              </svg>
              <span className="text-sm sm:text-base">{success}</span>
            </div>
          )}

          {/* Error Message */}
          {error && (
            <div className="mb-4 p-3 sm:p-4 bg-red-50 border border-red-200 text-red-700 rounded-lg flex items-center gap-2 animate-fadeIn">
              <svg
                className="w-5 h-5 flex-shrink-0"
                fill="currentColor"
                viewBox="0 0 20 20"
              >
                <path
                  fillRule="evenodd"
                  d="M10 18a8 8 0 100-16 8 8 0 000 16zM8.707 7.293a1 1 0 00-1.414 1.414L8.586 10l-1.293 1.293a1 1 0 101.414 1.414L10 11.414l1.293 1.293a1 1 0 001.414-1.414L11.414 10l1.293-1.293a1 1 0 00-1.414-1.414L10 8.586 8.707 7.293z"
                  clipRule="evenodd"
                />
              </svg>
              <span className="text-sm sm:text-base">{error}</span>
            </div>
          )}

          <form
            onSubmit={handleSubmit}
            className="bg-white rounded-lg shadow-sm p-3 sm:p-4 md:p-5 lg:p-6 xl:p-7 space-y-3 sm:space-y-4 md:space-y-5"
          >
            {/* Platform Selection Custom Dropdown with Icons */}
            <div className="space-y-1.5">
              <label className="block text-xs sm:text-sm md:text-sm lg:text-base font-semibold text-gray-700">
                Platform <span className="text-red-500">*</span>
              </label>
              <div className="relative" ref={dropdownRef}>
                {/* Dropdown Button */}
                <button
                  type="button"
                  onClick={() => setIsOpen(!isOpen)}
                  className="w-full px-2.5 sm:px-3 md:px-4 py-2 sm:py-2.5 md:py-2.5 lg:py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent bg-white text-gray-700 flex items-center justify-between text-xs sm:text-sm md:text-base"
                >
                  <div className="flex items-center gap-2">
                    {getSelectedPlatform() ? (
                      <>
                        <img
                          src={getSelectedPlatform().icon}
                          alt={getSelectedPlatform().name}
                          className="w-3.5 h-3.5 sm:w-4 sm:h-4 md:w-5 md:h-5"
                        />
                        <span>{getSelectedPlatform().name}</span>
                      </>
                    ) : (
                      <span className="text-gray-500">Select a platform</span>
                    )}
                  </div>
                  <svg
                    className={`w-3.5 h-3.5 sm:w-4 sm:h-4 text-gray-400 transition-transform ${
                      isOpen ? "rotate-180" : ""
                    }`}
                    fill="none"
                    stroke="currentColor"
                    viewBox="0 0 24 24"
                  >
                    <path
                      strokeLinecap="round"
                      strokeLinejoin="round"
                      strokeWidth="2"
                      d="M19 9l-7 7-7-7"
                    />
                  </svg>
                </button>

                {/* Dropdown Menu */}
                {isOpen && (
                  <div className="absolute z-10 w-full mt-1 bg-white border border-gray-300 rounded-lg shadow-lg max-h-60 overflow-auto">
                    {platforms.map((platform) => (
                      <div
                        key={platform.value}
                        onClick={() => handleSelectPlatform(platform.value)}
                        className={`flex items-center gap-2 px-2.5 sm:px-3 md:px-4 py-2 sm:py-2.5 cursor-pointer hover:bg-blue-50 transition-colors text-xs sm:text-sm md:text-base ${
                          selectedPlatform === platform.value
                            ? "bg-blue-50 text-blue-600 font-medium"
                            : "text-gray-700"
                        }`}
                      >
                        <img
                          src={platform.icon}
                          alt={platform.name}
                          className="w-3.5 h-3.5 sm:w-4 sm:h-4 md:w-5 md:h-5"
                        />
                        <span>{platform.name}</span>
                        {selectedPlatform === platform.value && (
                          <svg
                            className="w-3.5 h-3.5 sm:w-4 sm:h-4 ml-auto text-blue-600"
                            fill="currentColor"
                            viewBox="0 0 20 20"
                          >
                            <path
                              fillRule="evenodd"
                              d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z"
                              clipRule="evenodd"
                            />
                          </svg>
                        )}
                      </div>
                    ))}
                  </div>
                )}
              </div>
            </div>

            {/* Business Management Name - Only show for Facebook */}
            {selectedPlatform === "facebook" && (
              <div className="space-y-1.5 animate-fadeIn">
                <label className="block text-xs sm:text-sm md:text-sm lg:text-base font-semibold text-gray-700">
                  Business Management Name{" "}
                  <span className="text-red-500">*</span>
                </label>
                <input
                  type="text"
                  placeholder="Enter business management name"
                  value={bmName}
                  onChange={(e) => setBmName(e.target.value)}
                  className="w-full px-2.5 sm:px-3 md:px-4 py-2 sm:py-2.5 md:py-2.5 lg:py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent text-xs sm:text-sm md:text-base"
                  required
                />
              </div>
            )}

            {/* Access Token */}
            {selectedPlatform && (
              <div className="space-y-1.5 animate-fadeIn">
                <label className="block text-xs sm:text-sm md:text-sm lg:text-base font-semibold text-gray-700">
                  Access Token <span className="text-red-500">*</span>
                </label>
                <input
                  type="text"
                  placeholder="Enter access token"
                  value={accessToken}
                  onChange={(e) => setAccessToken(e.target.value)}
                  className="w-full px-2.5 sm:px-3 md:px-4 py-2 sm:py-2.5 md:py-2.5 lg:py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent text-xs sm:text-sm md:text-base"
                  required
                />
              </div>
            )}

            {/* Accounts Section */}
            {selectedPlatform && (
              <div className="space-y-3 sm:space-y-4 pt-3 sm:pt-4 border-t border-gray-200 animate-fadeIn">
                <h2 className="text-sm sm:text-base md:text-lg font-semibold text-gray-800">
                  Account Details
                </h2>

                <div className="grid grid-cols-1 md:grid-cols-2 gap-3 sm:gap-4">
                  <div className="space-y-1.5">
                    <label className="block text-xs sm:text-sm md:text-sm lg:text-base font-semibold text-gray-700">
                      Account ID <span className="text-red-500">*</span>
                    </label>
                    <input
                      type="text"
                      placeholder="Enter account ID"
                      value={accountId}
                      onChange={(e) => setAccountId(e.target.value)}
                      className="w-full px-2.5 sm:px-3 md:px-4 py-2 sm:py-2.5 md:py-2.5 lg:py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent text-xs sm:text-sm md:text-base"
                      required
                    />
                  </div>

                  <div className="space-y-1.5">
                    <label className="block text-xs sm:text-sm md:text-sm lg:text-base font-semibold text-gray-700">
                      Account Label <span className="text-red-500">*</span>
                    </label>
                    <input
                      type="text"
                      placeholder="Enter account label"
                      value={accountLabel}
                      onChange={(e) => setAccountLabel(e.target.value)}
                      className="w-full px-2.5 sm:px-3 md:px-4 py-2 sm:py-2.5 md:py-2.5 lg:py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent text-xs sm:text-sm md:text-base"
                      required
                    />
                  </div>
                </div>
              </div>
            )}

            {/* Submit Button */}
            <div className="flex justify-end pt-2 sm:pt-3">
              <button
                type="submit"
                disabled={!selectedPlatform || isSubmitting}
                className="w-full sm:w-auto px-4 sm:px-6 md:px-8 py-2 sm:py-2.5 md:py-2.5 lg:py-3 bg-blue-600 text-white font-semibold rounded-lg transition-all duration-200 hover:bg-blue-700 disabled:bg-gray-300 disabled:cursor-not-allowed shadow-md hover:shadow-lg flex items-center justify-center gap-2 text-xs sm:text-sm md:text-base"
              >
                {isSubmitting ? (
                  <>
                    <svg
                      className="animate-spin h-4 w-4 text-white"
                      xmlns="http://www.w3.org/2000/svg"
                      fill="none"
                      viewBox="0 0 24 24"
                    >
                      <circle
                        className="opacity-25"
                        cx="12"
                        cy="12"
                        r="10"
                        stroke="currentColor"
                        strokeWidth="4"
                      />
                      <path
                        className="opacity-75"
                        fill="currentColor"
                        d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"
                      />
                    </svg>
                    <span>Adding...</span>
                  </>
                ) : (
                  <>
                    <svg
                      className="w-3.5 h-3.5 sm:w-4 sm:h-4"
                      fill="none"
                      stroke="currentColor"
                      viewBox="0 0 24 24"
                    >
                      <path
                        strokeLinecap="round"
                        strokeLinejoin="round"
                        strokeWidth="2"
                        d="M12 4v16m8-8H4"
                      />
                    </svg>
                    <span>Add Account</span>
                  </>
                )}
              </button>
            </div>
          </form>
        </div>
      </div>
    </div>
  );
}

export default Add_Acounts;
